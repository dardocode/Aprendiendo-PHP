function initializeSlider(sliderId) { const sliderContainer = document.querySelector(`#${sliderId}`); const sliderCards = sliderContainer.querySelectorAll('.slider-card'); const sliderArrows = sliderContainer.querySelectorAll('.slider-arrow'); let currentCardIndex = 0; function showCard(index) { sliderCards.forEach(card => card.classList.remove('active')); sliderCards[index].classList.add('active'); } function navigateSlider(event) { const arrowClicked = event.target.closest('.slider-arrow'); const arrowDirection = arrowClicked.classList.contains('left') ? -1 : 1; currentCardIndex = (currentCardIndex + arrowDirection + sliderCards.length) % sliderCards.length; showCard(currentCardIndex); } sliderArrows.forEach(arrow => { arrow.addEventListener('click', navigateSlider); }); showCard(currentCardIndex); } initializeSlider('slider1');
